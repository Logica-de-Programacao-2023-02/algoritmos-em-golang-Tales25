// Faça um algoritmo que leia três números inteiros e mostre a soma entre eles.

package main

import "fmt"

func main() {
	var n1 int
	var n2 int
	var n3 int
	var soma int
	fmt.Print("Primeiro número: ")
	fmt.Scan(&n1)
	fmt.Print("Segundo número: ")
	fmt.Scan(&n2)
	fmt.Print("Terceiro número: ")
	fmt.Scan(&n3)
	soma = n1 + n2 + n3
	fmt.Print(n1, "+", n2, "+", n3, "=", soma)
}
